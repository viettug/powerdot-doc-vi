\begin{slide}{Ví dụ về slide}
  Công thức Newton:
  \begin{equation}\label{binomium}
    (a+b)^n=\sum_{k=0}^n{n\choose k}a^{n-k}b^k
  \end{equation}
  Ta sẽ chứng minh (\ref{binomium}) trên bảng đen.\\
  \begin{itemize}
    \item Đây
    \begin{itemize}
      \item là
      \begin{itemize}
        \item danh
        \begin{itemize}
          \item sách
        \end{itemize}
        \item với
      \end{itemize}
      \item bảy
    \end{itemize}
    \item phần tử
  \end{itemize}
\end{slide}
